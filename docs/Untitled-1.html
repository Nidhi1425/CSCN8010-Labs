from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_squared_error, mean_absolute_error

# Function to run linear regression and return model parameters
def run_linear_regression(X, y):
    lin_reg = LinearRegression()
    lin_reg.fit(X, y)
    intercept = lin_reg.intercept_
    slope = lin_reg.coef_[0]
    y_pred = lin_reg.predict(X)
    mse = mean_squared_error(y, y_pred)
    mae = mean_absolute_error(y, y_pred)
    return intercept, slope, mse, mae, y_pred

# Preparing data
X_income = housing_data[['median_income']]
y = housing_data['median_house_value']

# Model 1: Median Income vs. Median House Value
intercept_income, slope_income, mse_income, mae_income, y_pred_income = run_linear_regression(X_income, y)

# Model 2: Population vs. Median House Value
X_population = housing_data[['population']]
intercept_pop, slope_pop, mse_pop, mae_pop, y_pred_pop = run_linear_regression(X_population, y)

# Model 3: Number of Households vs. Median House Value
X_households = housing_data[['households']]
intercept_hh, slope_hh, mse_hh, mae_hh, y_pred_hh = run_linear_regression(X_households, y)

results = pd.DataFrame({
    'Model': ['Median Income'],
    'Intercept': [intercept_income],
    'Slope': [slope_income],
    'MSE': [mse_income],
    'MAE': [mae_income]
})
print(results)
